문제

영어 단어를 입력하면 영한 사전을 열고 그 뜻을 찾아서 출력하는 문제입니다 실행순서를 클릭하세요 
--------- 파일 dictionary.txt의 내용 --------- 
apple의 뜻은 사과입니다 
banana의 뜻은 바나나입니다 
carrot의 뜻은 당근입니다 

영어단어를 입력하세요: banana
사전이름을 입력하세요: C:\Users\user\Downloads\dictionary.txt
banana의 뜻은 바나나입니다

프로그램 코드

	#include <stdio.h>

	#include <string.h>


1	int main() {

		printf( "사전이름을 입력하세요: " );

		char filename[1024] = "";

2		scanf( "%s", filename );

		FILE *dictionary = NULL;

3		if ( ( dictionary = fopen( filename, "rt" ) ) == NULL ) {

			printf( "[오류] 파일을 열 수 없습니다!" );

			return -1;

		}

		

		printf( "영어단어를 입력하세요: " );

		char word[1024] = "";

4		scanf( "%s", word );


5		for ( char meaning[1024] = ""; fgets( meaning, 1024, dictionary ) != NULL; ) {

6			if ( !strncmp( word,  meaning, strlen( word ) ) ) {

				printf( "%s\n", meaning );

7				break;

			}

		}


8		fclose( dictionary );


9		return 0;

	}
