문제

게임 스코어를 입력받아 게임 랭킹 파일을 업데이트하는 문제입니다 실행순서를 클릭하세요 
--------- 파일 ranking.txt의 내용 --------- 
내가짱	8579214
구름위	3674159
이기자	1578236

게임랭킹 파일이름을 입력하세요: C:\Users\user\Downloads\ranking.txt
게임 아이디를 입력하세요: 홍길동
게임 스코어를 입력하세요: 4157967
1) 내가짱 8579214
2) 홍길동 4157967
3) 구름위 3674159
4) 이기자 1578236

프로그램 코드

	#include <stdio.h>

	#include <string.h>

	#include <stdlib.h>


1	int main() {

		char id[5][2][32] = { { "", "" }, { "", "" }, { "", "" }, { "", "" }, { "", "" } };


		printf( "게임랭킹 파일이름을 입력하세요: " );

		char filename[1024] = "";

2		scanf( "%s", filename );

		FILE *inRankingFile = NULL;

3		if ( ( inRankingFile = fopen( filename, "rt" ) ) == NULL ) {

			printf("[오류] 게임랭킹 파일을 열 수 없습니다!");

			return -1;

		}


		int index = 0;

4		for (; !feof( inRankingFile ); index++ ) {

5			if ( fscanf( inRankingFile, "%s\t%s\n", id[index][0], id[index][1] ) < 2 ) {

				printf( "(오류) 게임랭킹 내용을 읽어올 수 없습니다." );

6				break;

			}

		}


7		fclose( inRankingFile );


		printf( "게임 아이디를 입력하세요: " );

		scanf( "%s", id[index][0] );

		printf( "게임 스코어를 입력하세요: " );

8		scanf( "%s", id[index][1] );


		for ( int last = index; last >= 0; last-- ) {

			for ( index = 0; index < last; index++ ) {

				if ( atoi( id[index][1] ) < atoi( id[index+1][1] ) ) {

					char temp[32];

					strcpy( temp, id[index][0] );

					strcpy( id[index][0], id[index+1][0] );

					strcpy( id[index+1][0], temp );

					strcpy( temp, id[index][1] );

					strcpy( id[index][1], id[index+1][1] );

					strcpy( id[index+1][1], temp );

				}

			}

		}


		FILE* outRankingFile = NULL;

9		if ( ( outRankingFile = fopen( filename, "wt" ) ) == NULL ) {

			printf( "[오류] 게임랭킹 파일을 열 수 없습니다!" );

			return -1;

		}


10		for ( index = 0; strcmp( id[index][0], "" ); index++ ) {

			fprintf( outRankingFile, "%s\t%s\n", id[index][0], id[index][1] );

11			printf( "%d) %s\t%s\n", index+1, id[index][0], id[index][1] );

		}


12		fclose( outRankingFile );


13		return 0;

	}
